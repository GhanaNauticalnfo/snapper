Show tables in test database

kubectl exec -it $(kubectl get pods -n test -l app=snapper-postgres -o name) -n test -- \
  psql -U $(kubectl get secret snapper-postgres-secret -n test -o jsonpath='{.data.POSTGRES_USER}' | base64 --decode) \
  -d $(kubectl get secret snapper-postgres-secret -n test -o jsonpath='{.data.POSTGRES_DB}' | base64 --decode) \
  -c "\dt"  # List tables
